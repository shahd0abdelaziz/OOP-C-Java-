#include<iostream>                  
using namespace std;
class Employee {
private:
	string name;
	int id;
	int age;
public:
	Employee() {
		cin >> name >> id >> age;
	}
	void print() {
		cout << name << " " << id << " " << age;
	}
};

int main() {
	Employee ob;
	ob.print();
}
